セキュリティ（簡易）
まずは匿名ログイン必須にした最小ルールを同梱しています。運用では「世帯コードを知っているだけで読める」状態になり得るため、 共同生活者以外に漏れないコードにしてください（必要なら、より堅牢なルール案も作れます）。 
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /households/{hid}/expenses/{docId} {
      allow read, write: if request.auth != null;
    }
    match /households/{hid}/members/{uid} {
      allow read, write: if request.auth != null && request.auth.uid == uid;
    }
  }
}
